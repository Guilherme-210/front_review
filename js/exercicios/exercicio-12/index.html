Exercício 12: Pilha de Cartas

Descrição

Este exercício simula um baralho de cartas, permitindo gerenciar as cartas através de um menu interativo.

Exemplo de Código JavaScript

let baralho = [];

function mostrarMenu() {
    let opcoes = "Quantidade de cartas no baralho: " + baralho.length + "\n";
    opcoes += "Escolha uma opção:\n";
    opcoes += "1. Adicionar uma carta\n";
    opcoes += "2. Puxar uma carta\n";
    opcoes += "3. Sair";
    return prompt(opcoes);
}

function adicionarCarta() {
    let carta = prompt("Digite o nome da carta:");
    if (carta) {
        baralho.push(carta);
        alert(carta + " adicionada ao baralho.");
    }
}

function puxarCarta() {
    if (baralho.length > 0) {
        let carta = baralho.pop();
        alert("Carta puxada: " + carta);
    } else {
        alert("Nenhuma carta no baralho.");
    }
}

let opcao;
do {
    opcao = mostrarMenu();
    switch (opcao) {
        case "1":
            adicionarCarta();
            break;
        case "2":
            puxarCarta();
            break;
    }
} while (opcao !== "3");

Explicação

O programa permite gerenciar um baralho de cartas com as seguintes funcionalidades:

  • Adicionar uma carta ao topo do baralho.
  • Puxar a carta do topo do baralho e exibir seu nome.
  • Mostrar a quantidade de cartas no baralho no menu.
  • A opção de sair encerra o programa.
