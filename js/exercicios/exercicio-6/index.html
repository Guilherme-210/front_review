Exercício 6: Visitando novas cidades

Requisitos

  • Pedir o nome do turista.
  • Perguntar se ele já visitou alguma cidade.
  • Se sim, pedir o nome da cidade e armazená-la.
  • Repetir a pergunta até que o turista responda que não visitou mais cidades.
  • Exibir o nome do turista, o número de cidades visitadas e a lista de cidades no final.

Exemplo de Código

// Exemplo de código em JavaScript
let nomeTurista = prompt("Qual é o seu nome?");
let cidades = [];
let visitou = prompt("Você já visitou alguma cidade? (Sim/Não)");

while (visitou.toLowerCase() === "sim") {
    let cidade = prompt("Qual é o nome da cidade?");
    cidades.push(cidade);
    visitou = prompt("Você visitou mais alguma cidade? (Sim/Não)");
}

let mensagem = "Turista: " + nomeTurista + "\n";
mensagem += "Número de cidades visitadas: " + cidades.length + "\n";
mensagem += "Cidades: " + cidades.join(", ");

alert(mensagem);

Resolução

const turista = prompt('E aí, turista! Qual é o seu nome?')
let cidades = ''
let contagem = 0

let continuar = prompt('Você visitou alguma cidade? (Sim/Não)').toLowerCase();

while (continuar === 'sim' || continuar === 's') {
  let cidade = prompt('Qual é o nome da cidade visitada?')
  cidades += ` - ${cidade} \n`
  contagem++
  continuar = prompt('Você visitou alguma outra cidade? (Sim/Não)').toLowerCase();
}
alert(`Turista: ${turista} 
  \nQuantidade de cidades visitadas: ${contagem} 
  \nCidades visitadas: \n ${cidades}`)
