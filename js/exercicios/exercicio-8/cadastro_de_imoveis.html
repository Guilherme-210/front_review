Exercício 13: Cadastro de imóveis

Descrição

Desenvolver um programa em JavaScript para cadastro de imóveis, permitindo que o usuário salve e visualize as informações dos imóveis cadastrados.

Exemplo de Código JavaScript

let imoveis = [];

function cadastrarImovel() {
    let proprietario = prompt("Nome do proprietário:");
    let quartos = prompt("Número de quartos:");
    let banheiros = prompt("Número de banheiros:");
    let garagem = confirm("Possui garagem?");

    let imovel = {
        proprietario: proprietario,
        quartos: Number(quartos),
        banheiros: Number(banheiros),
        garagem: garagem
    };

    imoveis.push(imovel);
    alert("Imóvel cadastrado com sucesso! Total de imóveis: " + imoveis.length);
}

function mostrarImoveis() {
    if (imoveis.length === 0) {
        alert("Nenhum imóvel cadastrado.");
        return;
    }
    
    let listaImoveis = "Imóveis cadastrados:\n";
    imoveis.forEach((imovel, index) => {
        listaImoveis += (index + 1) + ". " + 
            "Proprietário: " + imovel.proprietario + ", " + 
            "Quartos: " + imovel.quartos + ", " + 
            "Banheiros: " + imovel.banheiros + ", " + 
            "Garagem: " + (imovel.garagem ? "Sim" : "Não") + "\n";
    });
    alert(listaImoveis);
}

// Exemplo de uso
cadastrarImovel();
mostrarImoveis();

Explicação

O programa permite ao usuário cadastrar e visualizar imóveis utilizando arrays e objetos:

  • Os imóveis são armazenados em um array.
  • Cada imóvel é representado como um objeto com propriedades: proprietário, quartos, banheiros e garagem.
  • Funções interativas permitem o cadastro e a visualização dos imóveis.
  • Prática dos conceitos de arrays e objetos em JavaScript.
